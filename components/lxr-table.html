<template>
  <div class="h-[200px] w-full">
    <lxr-table
      :columns="columns"
      :rows="sortedRows"
      :sort="sort"
      @sort="sort = $event"/>
  </div>
</template>

<script lang="ts" setup>
import { ref, unref, computed } from 'vue'
import { LxrTable, LxrColumn, LxrTableSort } from 'lxr-ui'
import Chance from 'chance'

const chance = new Chance()

interface Row {
  name: string
  jobTitle: string
  country: string
}

const generateRow = () => ({
  name: chance.name(),
  jobTitle: chance.profession(),
  country: chance.country({ full: true })
})

const generateRows = (count: number) => [...Array(count).keys()].map(generateRow)

const columns: LxrColumn[] = [
  { key: 'name', label: 'Name', sortable: true },
  { key: 'jobTitle', label: 'Job Title', sortable: true },
  { key: 'country', label: 'Country', sortable: true }
]

const sort = ref<null | LxrTableSort>(null)
const rows = generateRows(50)

const sortedRows = computed(() => {
  if (unref(sort) === null) return rows
  const [[key, asc]] = Object.entries(unref(sort))
  const sortingFactor = asc ? 1 : -1
  return [...rows].sort(({ [key as keyof Row]: A }, { [key as keyof Row]: B }) => A > B ? sortingFactor : A < B ? -sortingFactor : 0)
})
</script>

<template>
  <div class="h-[200px]">
    <lxr-table
      :columns="columns"
      :rows="rows"
      threshold="30%"
      :fetch-more-callback="fetchMoreCallback" />
  </div>
</template>

<script lang="ts" setup>
import { ref } from 'vue'
import { LxrTable, LxrColumn, LxrTableSort } from 'lxr-ui'
import Chance from 'chance'

export interface Row {
  name: string
  jobTitle: string
  country: string
}

const chance = new Chance()

const generateRow = (): Row => ({
  name: chance.name(),
  jobTitle: chance.profession(),
  country: chance.country({ full: true })
})

const generateRows = (count: number) => [...Array(count).keys()].map(generateRow)

const columns: LxrColumn[] = [
  { key: 'name', label: 'Name' },
  { key: 'jobTitle', label: 'Job Title' }
]

const rows = ref(generateRows(50))

const fetchMoreCallback = async () => {
  return new Promise<void>(resolve => {
    setTimeout(() => {
      rows.value.push(...generateRows(50))
      resolve()
    }, 1000)
  })
}
</script>

# Props

NameTypeRequiredDefaultDescription
columnsLxrColumn[]yestable column definition
rowsany[]yestable rows
sortnull | Record<string, boolean>noSorting key
thresholdstring | numbernoThe threshold distance from the bottom of the content to call the infinite output event when scrolled. Value can be a number (pixels, from bottom) or string (in "px" or "%"). Defaults to "15%".
fetchMoreCallback() => Promise<void>noOptional callback that gets triggered when user scrolls below the defined threshold level.
tableClassesstringnoOptional css classes to be applied to the table element.
rowClassesstringnoOptional css classes to be applied to all table rows.

# Events

NameCallbackDescription
sort(null | Record<string, boolean>) => voidTriggered on user UI interaction with column sort controls. Callback argument is either null (sorting disabled) or object having as key the sorting key and value a boolean refering the sorting direction (asc).
